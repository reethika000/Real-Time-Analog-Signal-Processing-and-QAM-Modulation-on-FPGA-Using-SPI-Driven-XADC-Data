//SPI INTERFACE CODE
`timescale 1ns / 1ps
module spi_interface(clk,sclk,cs,do_out,drdy_out,shift_reg,miso,count);
input clk,sclk;
input drdy_out,cs;
input [15:0]do_out;
output reg [15:0]shift_reg;
output reg miso;
output reg [3:0]count;

always@(negedge clk)
if(drdy_out)
begin
shift_reg<=do_out;
count<=0;
miso<=0;
end

always@(negedge sclk)
if(!cs)
begin
miso<=shift_reg[15];
shift_reg<=shift_reg<<1;
count<=count+1;
end
endmodule

    reg signed [3:0] i_level, q_level;

        input [1:0] bits;
        begin
            case (bits)
                2'b00: map_qam_bits = -3;
                2'b01: map_qam_bits = -1;
                2'b10: map_qam_bits = +1;
                2'b11: map_qam_bits = +3;
            endcase
        end
    endfunction

       data_out <= data_in;

       i_level <= data_in;
end
